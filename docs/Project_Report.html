Disconnected Systems: Different departments often use separate systems that don't communicate with each other. For example, the billing department may use one system while the appointment scheduling uses another, requiring manual data transfer and increasing the likelihood of errors.

public function generateBill($data) {

            str_pad(rand(1, 9999), 4, '0', STR_PAD_LEFT);
        
        $totalAmount = 0;
        foreach ($data['items'] as $item) {
            $totalAmount += $item['quantity'] * $item['unit_price'];
        }
        
        $tax = $totalAmount * 0.10; // 10% tax
        $finalAmount = $totalAmount + $tax;
        
        $query = "INSERT INTO billing (
            bill_number, patient_id, appointment_id,
            total_amount, payment_status
        ) VALUES (
            :bill_number, :patient_id, :appointment_id,
            :total_amount, 'pending'
        )";

        $stmt->bindParam(':bill_number', $billNumber);

        $stmt->bindParam(':appointment_id', $data['appointment_id']);
        $stmt->bindParam(':total_amount', $finalAmount);
        
        if ($stmt->execute()) {
            $billingId = $this->conn->lastInsertId();
            
            // Insert billing items
            foreach ($data['items'] as $item) {
                $this->addBillingItem($billingId, $item);
            }
            
            return [
                'success' => true,
                'message' => 'Bill generated successfully',
                'bill_number' => $billNumber,

    } catch (PDOException $e) {
        return [
            'success' => false,
            'message' => 'Database error: ' . $e->getMessage()
        ];
    }
}
